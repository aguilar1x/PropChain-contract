703        pub fn change_admin(&mut self, new_admin: AccountId) -> Result<(), Error> {
704            let caller = self.env().caller();
705            if caller != self.admin {
706                return Err(Error::Unauthorized);
707            }
708
709            let old_admin = self.admin;
710            self.admin = new_admin;
711
712            // Emit enhanced admin changed event
713
714            let transaction_hash: Hash = [0u8; 32].into();
715            self.env().emit_event(AdminChanged {
716                old_admin,
717                new_admin,
718                event_version: 1,
719                timestamp: self.env().block_timestamp(),
720                block_number: self.env().block_number(),
721                transaction_hash,
722                changed_by: caller,
723            });
724
725            Ok(())
726        }
