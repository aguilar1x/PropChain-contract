1use anyhow::{Context, Result};
2use clap::{Parser, Subcommand};
3use colored::*;
4use serde::{Deserialize, Serialize};
5use std::fs;
6use std::process::Command;
7use walkdir::WalkDir;
8
9#[derive(Parser)]
10#[command(author, version, about, long_about = None)]
11struct Cli {
12    #[command(subcommand)]
13    command: Commands,
14}
15
16#[derive(Subcommand)]
17enum Commands {
18    /// Run full security audit
19    Audit {
20        /// Generate a report file
21        #[arg(short, long)]
22        report: Option<String>,
23    },
24}
25
26#[derive(Serialize, Deserialize, Debug, Default)]
27struct SecurityReport {
28    timestamp: String,
29    score: u32,
30    static_analysis: StaticAnalysisResults,
31    dependency_scan: DependencyScanResults,
32    code_quality: CodeQualityResults,
33}
34
35#[derive(Serialize, Deserialize, Debug, Default)]
36struct StaticAnalysisResults {
37    clippy_warnings: usize,
38    clippy_errors: usize,
39    complexity_warnings: usize,
40    unsafe_blocks: usize,
41    todos_found: usize,
42}
43
44#[derive(Serialize, Deserialize, Debug, Default)]
45struct DependencyScanResults {
46    vulnerabilities: usize,
47    warnings: usize,
48}
49
50#[derive(Serialize, Deserialize, Debug, Default)]
51struct CodeQualityResults {
52    complexity_score: u32,
53    files_scanned: usize,
54}
55
56fn main() -> Result<()> {
57    let cli = Cli::parse();
58
59    match cli.command {
60        Commands::Audit { report } => {
61            println!("{}", "Starting Security Audit Pipeline...".blue().bold());
62            
63            let mut audit_report = SecurityReport {
64                timestamp: chrono::Utc::now().to_rfc3339(),
65                ..Default::default()
66            };
67
68            // 1. Static Analysis (Clippy)
69            println!("{}", "Running Static Analysis (Clippy)...".yellow());
70            let clippy_output = Command::new("cargo")
71                .args(["clippy", "--message-format=json", "--all-targets", "--all-features"])
72                .output()
73                .context("Failed to run cargo clippy")?;
74            
75            // Parse clippy output (simplified)
76            let output_str = String::from_utf8_lossy(&clippy_output.stdout);
77            for line in output_str.lines() {
78                if let Ok(json) = serde_json::from_str::<serde_json::Value>(line) {
79                    if let Some(level) = json.get("level").and_then(|l| l.as_str()) {
80                        match level {
81                            "warning" => {
82                                audit_report.static_analysis.clippy_warnings += 1;
83                                if let Some(message) = json.get("message").and_then(|m| m.as_object()) {
84                                    if let Some(code) = message.get("code").and_then(|c| c.as_object()) {
85                                        if let Some(code_str) = code.get("code").and_then(|s| s.as_str()) {
86                                            if code_str.contains("complexity") {
87                                                audit_report.static_analysis.complexity_warnings += 1;
88                                            }
89                                        }
90                                    }
91                                }
92                            },
93                            "error" => audit_report.static_analysis.clippy_errors += 1,
94                            _ => {}
95                        }
96                    }
97                }
98            }
99
100            // 2. Custom Linter (Unsafe & TODOs)
101            println!("{}", "Running Custom Rust Security Linters...".yellow());
102            for entry in WalkDir::new(".").into_iter().filter_map(|e| e.ok()) {
103                if entry.path().extension().map_or(false, |ext| ext == "rs") {
104                    audit_report.code_quality.files_scanned += 1;
105                    let content = fs::read_to_string(entry.path()).unwrap_or_default();
106                    
107                    audit_report.static_analysis.unsafe_blocks += content.matches("unsafe {").count();
108                    audit_report.static_analysis.todos_found += content.matches("TODO").count();
109                    audit_report.static_analysis.todos_found += content.matches("FIXME").count();
110                }
111            }
112
113            // 3. Dependency Scan (cargo audit)
114            println!("{}", "Running Dependency Vulnerability Scanning...".yellow());
115            // Check if cargo-audit is installed
116            if Command::new("cargo").args(["audit", "--version"]).output().is_ok() {
117                let audit_cmd = Command::new("cargo")
118                    .args(["audit", "--json"])
119                    .output();
120                
121                if let Ok(output) = audit_cmd {
122                    let output_str = String::from_utf8_lossy(&output.stdout);
123                    if let Ok(json) = serde_json::from_str::<serde_json::Value>(&output_str) {
124                         if let Some(vulns) = json.get("vulnerabilities").and_then(|v| v.as_object()) {
125                             if let Some(list) = vulns.get("list").and_then(|l| l.as_array()) {
126                                 audit_report.dependency_scan.vulnerabilities = list.len();
127                             }
128                         }
129                         if let Some(warnings) = json.get("warnings").and_then(|w| w.as_object()) {
130                            // Count warnings if structure matches, otherwise simplified
131                             audit_report.dependency_scan.warnings = warnings.len();
132                         }
133                    }
134                } else {
135                    println!("{}", "cargo audit failed to run".red());
136                }
137            } else {
138                println!("{}", "cargo-audit not found. Skipping...".red());
139            }
140
141            // Calculate Score
142            let mut score: u32 = 100;
143            score = score.saturating_sub((audit_report.static_analysis.clippy_errors * 10) as u32);
144            score = score.saturating_sub((audit_report.static_analysis.clippy_warnings * 2) as u32);
145            score = score.saturating_sub((audit_report.static_analysis.complexity_warnings * 5) as u32);
146            score = score.saturating_sub((audit_report.static_analysis.unsafe_blocks * 5) as u32);
147            score = score.saturating_sub((audit_report.dependency_scan.vulnerabilities * 20) as u32);
148            
149            audit_report.score = score;
150
151            println!("{}", "Audit Complete!".green().bold());
152            println!("Security Score: {}/100", score);
153            println!("Clippy Issues: {} errors, {} warnings ({} complexity)", 
154                audit_report.static_analysis.clippy_errors, 
155                audit_report.static_analysis.clippy_warnings,
156                audit_report.static_analysis.complexity_warnings
157            );
158            println!("Unsafe Blocks: {}", audit_report.static_analysis.unsafe_blocks);
159            println!("Vulnerabilities: {}", audit_report.dependency_scan.vulnerabilities);
160
161            if let Some(path) = report {
162                let report_json = serde_json::to_string_pretty(&audit_report)?;
163                fs::write(path, report_json)?;
164                println!("Report saved to file.");
165            }
166        }
167    }
168    Ok(())
169}
