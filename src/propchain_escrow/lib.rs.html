277        /// Create a new escrow with advanced features
278        #[ink(message)]
279        pub fn create_escrow_advanced(
280            &mut self,
281            property_id: u64,
282            amount: u128,
283            buyer: AccountId,
284            seller: AccountId,
285            participants: Vec<AccountId>,
286            required_signatures: u8,
287            release_time_lock: Option<u64>,
288        ) -> Result<u64, Error> {
289            let caller = self.env().caller();
290            
291            // Validate configuration
292            if required_signatures == 0 || participants.is_empty() {
293                return Err(Error::InvalidConfiguration);
294            }
295
296            if required_signatures as usize > participants.len() {
297                return Err(Error::InvalidConfiguration);
298            }
299
300            self.escrow_count += 1;
301            let escrow_id = self.escrow_count;
302
303            // Create escrow data
304            let escrow_data = EscrowData {
305                id: escrow_id,
306                property_id,
307                buyer,
308                seller,
309                amount,
310                deposited_amount: 0,
311                status: EscrowStatus::Created,
312                created_at: self.env().block_timestamp(),
313                release_time_lock,
314                participants: participants.clone(),
315            };
316
317            self.escrows.insert(&escrow_id, &escrow_data);
318
319            // Set up multi-sig configuration
320            let multi_sig_config = MultiSigConfig {
321                required_signatures,
322                signers: participants.clone(),
323            };
324            self.multi_sig_configs.insert(&escrow_id, &multi_sig_config);
325
326            // Initialize empty collections
327            self.documents.insert(&escrow_id, &Vec::<DocumentHash>::new());
328            self.conditions.insert(&escrow_id, &Vec::<Condition>::new());
329            self.condition_counters.insert(&escrow_id, &0);
330            self.audit_logs.insert(&escrow_id, &Vec::<AuditEntry>::new());
331
332            // Add audit entry
333            self.add_audit_entry(
334                escrow_id,
335                caller,
336                "EscrowCreated".to_string(),
337                format!("Property: {}, Amount: {}", property_id, amount),
338            );
339
340            self.env().emit_event(EscrowCreated {
341                escrow_id,
342                property_id,
343                buyer,
344                seller,
345                amount,
346            });
347
348            Ok(escrow_id)
349        }

351        /// Deposit funds to escrow
352        #[ink(message, payable)]
353        pub fn deposit_funds(&mut self, escrow_id: u64) -> Result<(), Error> {
354            let caller = self.env().caller();
355            let transferred = self.env().transferred_value();
356            
357            let mut escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
358
359            // Verify escrow is in correct state
360            if escrow.status != EscrowStatus::Created && escrow.status != EscrowStatus::Funded {
361                return Err(Error::InvalidStatus);
362            }
363
364            // Update deposited amount
365            escrow.deposited_amount += transferred;
366
367            // Check if fully funded
368            if escrow.deposited_amount >= escrow.amount {
369                escrow.status = EscrowStatus::Active;
370            } else {
371                escrow.status = EscrowStatus::Funded;
372            }
373
374            self.escrows.insert(&escrow_id, &escrow);
375
376            // Add audit entry
377            self.add_audit_entry(
378                escrow_id,
379                caller,
380                "FundsDeposited".to_string(),
381                format!("Amount: {}", transferred),
382            );
383
384            self.env().emit_event(FundsDeposited {
385                escrow_id,
386                amount: transferred,
387                depositor: caller,
388            });
389
390            Ok(())
391        }

393        /// Release funds with multi-signature approval
394        #[ink(message)]
395        pub fn release_funds(&mut self, escrow_id: u64) -> Result<(), Error> {
396            let caller = self.env().caller();
397            let escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
398
399            // Check status
400            if escrow.status != EscrowStatus::Active {
401                return Err(Error::InvalidStatus);
402            }
403
404            // Check for active dispute
405            if let Some(dispute) = self.disputes.get(&escrow_id) {
406                if !dispute.resolved {
407                    return Err(Error::DisputeActive);
408                }
409            }
410
411            // Check time lock
412            if let Some(time_lock) = escrow.release_time_lock {
413                if self.env().block_timestamp() < time_lock {
414                    return Err(Error::TimeLockActive);
415                }
416            }
417
418            // Check all conditions are met
419            if !self.check_all_conditions_met(escrow_id)? {
420                return Err(Error::ConditionsNotMet);
421            }
422
423            // Check multi-sig threshold
424            if !self.check_signature_threshold(escrow_id, ApprovalType::Release)? {
425                return Err(Error::SignatureThresholdNotMet);
426            }
427
428            // Transfer funds to seller
429            if self.env().transfer(escrow.seller, escrow.deposited_amount).is_err() {
430                return Err(Error::InsufficientFunds);
431            }
432
433            // Update status
434            let mut updated_escrow = escrow.clone();
435            updated_escrow.status = EscrowStatus::Released;
436            self.escrows.insert(&escrow_id, &updated_escrow);
437
438            // Add audit entry
439            self.add_audit_entry(
440                escrow_id,
441                caller,
442                "FundsReleased".to_string(),
443                format!("Amount: {} to seller", escrow.deposited_amount),
444            );
445
446            self.env().emit_event(FundsReleased {
447                escrow_id,
448                amount: escrow.deposited_amount,
449                recipient: escrow.seller,
450            });
451
452            Ok(())
453        }

680        /// Sign approval for release or refund
681        #[ink(message)]
682        pub fn sign_approval(&mut self, escrow_id: u64, approval_type: ApprovalType) -> Result<(), Error> {
683            let caller = self.env().caller();
684            let _escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
685            let config = self.multi_sig_configs.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
686
687            // Check if caller is a valid signer
688            if !config.signers.contains(&caller) {
689                return Err(Error::Unauthorized);
690            }
691
692            // Check if already signed
693            let sig_key = (escrow_id, approval_type.clone(), caller);
694            if self.signatures.get(&sig_key).unwrap_or(false) {
695                return Err(Error::AlreadySigned);
696            }
697
698            // Add signature
699            self.signatures.insert(&sig_key, &true);
700
701            // Update signature count
702            let count_key = (escrow_id, approval_type.clone());
703            let current_count = self.signature_counts.get(&count_key).unwrap_or(0);
704            self.signature_counts.insert(&count_key, &(current_count + 1));
705
706            // Add audit entry
707            self.add_audit_entry(
708                escrow_id,
709                caller,
710                "SignatureAdded".to_string(),
711                format!("Approval type: {:?}", approval_type),
712            );
713
714            self.env().emit_event(SignatureAdded {
715                escrow_id,
716                approval_type,
717                signer: caller,
718            });
719
720            Ok(())
721        }

723        /// Raise a dispute
724        #[ink(message)]
725        pub fn raise_dispute(&mut self, escrow_id: u64, reason: String) -> Result<(), Error> {
726            let caller = self.env().caller();
727            let escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
728
729            // Only buyer or seller can raise dispute
730            if caller != escrow.buyer && caller != escrow.seller {
731                return Err(Error::Unauthorized);
732            }
733
734            // Check if dispute already exists
735            if let Some(existing_dispute) = self.disputes.get(&escrow_id) {
736                if !existing_dispute.resolved {
737                    return Err(Error::DisputeActive);
738                }
739            }
740
741            let dispute = DisputeInfo {
742                escrow_id,
743                raised_by: caller,
744                reason: reason.clone(),
745                raised_at: self.env().block_timestamp(),
746                resolved: false,
747                resolution: None,
748            };
749
750            self.disputes.insert(&escrow_id, &dispute);
751
752            // Update escrow status
753            let mut updated_escrow = escrow;
754            updated_escrow.status = EscrowStatus::Disputed;
755            self.escrows.insert(&escrow_id, &updated_escrow);
756
757            // Add audit entry
758            self.add_audit_entry(
759                escrow_id,
760                caller,
761                "DisputeRaised".to_string(),
762                format!("Reason: {}", reason),
763            );
764
765            self.env().emit_event(DisputeRaised {
766                escrow_id,
767                raised_by: caller,
768                reason,
769            });
770
771            Ok(())
772        }

774        /// Resolve dispute (admin only)
775        #[ink(message)]
776        pub fn resolve_dispute(&mut self, escrow_id: u64, resolution: String) -> Result<(), Error> {
777            let caller = self.env().caller();
778
779            // Only admin can resolve disputes
780            if caller != self.admin {
781                return Err(Error::Unauthorized);
782            }
783
784            let mut dispute = self.disputes.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
785            dispute.resolved = true;
786            dispute.resolution = Some(resolution.clone());
787            self.disputes.insert(&escrow_id, &dispute);
788
789            // Update escrow status back to Active
790            let mut escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
791            escrow.status = EscrowStatus::Active;
792            self.escrows.insert(&escrow_id, &escrow);
793
794            // Add audit entry
795            self.add_audit_entry(
796                escrow_id,
797                caller,
798                "DisputeResolved".to_string(),
799                format!("Resolution: {}", resolution),
800            );
801
802            self.env().emit_event(DisputeResolved {
803                escrow_id,
804                resolution,
805            });
806
807            Ok(())
808        }

810        /// Emergency override (admin only)
811        #[ink(message)]
812        pub fn emergency_override(&mut self, escrow_id: u64, release_to_seller: bool) -> Result<(), Error> {
813            let caller = self.env().caller();
814
815            // Only admin can perform emergency override
816            if caller != self.admin {
817                return Err(Error::Unauthorized);
818            }
819
820            let escrow = self.escrows.get(&escrow_id).ok_or(Error::EscrowNotFound)?;
821
822            let recipient = if release_to_seller {
823                escrow.seller
824            } else {
825                escrow.buyer
826            };
827
828            // Transfer funds
829            if self.env().transfer(recipient, escrow.deposited_amount).is_err() {
830                return Err(Error::InsufficientFunds);
831            }
832
833            // Update status
834            let mut updated_escrow = escrow.clone();
835            updated_escrow.status = if release_to_seller {
836                EscrowStatus::Released
837            } else {
838                EscrowStatus::Refunded
839            };
840            self.escrows.insert(&escrow_id, &updated_escrow);
841
842            // Add audit entry
843            self.add_audit_entry(
844                escrow_id,
845                caller,
846                "EmergencyOverride".to_string(),
847                format!("Funds sent to: {:?}", recipient),
848            );
849
850            self.env().emit_event(EmergencyOverride {
851                escrow_id,
852                admin: caller,
853            });
854
855            Ok(())
856        }
