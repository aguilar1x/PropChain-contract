161            // Check price alerts

163
164            // Emit event
165            self.env().emit_event(ValuationUpdated {
166                property_id,
167                valuation: valuation.valuation,
168                confidence_score: valuation.confidence_score,
169                timestamp: self.env().block_timestamp(),
170            });

177        pub fn update_valuation_from_sources(&mut self, property_id: u64) -> Result<(), OracleError> {
178            // Collect prices from all active sources
179            let prices = self.collect_prices_from_sources(property_id)?;
180
181            if prices.len() < self.min_sources_required as usize {
182                return Err(OracleError::InsufficientSources);
183            }
184
185            // Aggregate prices with outlier detection
186            let aggregated_price = self.aggregate_prices(&prices)?;
187            let confidence_score = self.calculate_confidence_score(&prices)?;
188
189            let valuation = PropertyValuation {
190                property_id,
191                valuation: aggregated_price,
192                confidence_score,
193                sources_used: prices.len() as u32,
194                last_updated: self.env().block_timestamp(),
195                valuation_method: ValuationMethod::MarketData,
196            };
197
198            self.update_property_valuation(property_id, valuation)
199        }

355        pub fn aggregate_prices(&self, prices: &[PriceData]) -> Result<u128, OracleError> {
356            if prices.is_empty() {
357                return Err(OracleError::InsufficientSources);
358            }
359
360            // Remove outliers
361            let filtered_prices = self.filter_outliers(prices);
362
363            if filtered_prices.is_empty() {
364                return Err(OracleError::InsufficientSources);
365            }
366
367            // Weighted average based on source weights
368            let mut total_weighted_price = 0u128;
369            let mut total_weight = 0u32;
370
371            for price_data in &filtered_prices {
372                let weight = self.get_source_weight(&price_data.source)?;
373                total_weighted_price += price_data.price * weight as u128;
374                total_weight += weight;
375            }
376
377            if total_weight == 0 {
378                return Err(OracleError::InvalidParameters);
379            }
380
381            Ok(total_weighted_price / total_weight as u128)
382        }

384        pub fn filter_outliers(&self, prices: &[PriceData]) -> Vec<PriceData> {
385            if prices.len() < 3 {
386                return prices.to_vec();
387            }
388
389            // Calculate mean
390            let sum: u128 = prices.iter().map(|p| p.price).sum();
391            let mean = sum / prices.len() as u128;
392
393            // Calculate standard deviation using fixed point arithmetic
394            let variance: u128 = prices.iter()
395                .map(|p| {
396                    let diff = if p.price > mean { p.price - mean } else { mean - p.price };
397                    diff * diff
398                })
399                .sum();
400
401            let variance_avg = variance / prices.len() as u128;
402            // Simple square root approximation
403            let mut std_dev = variance_avg;
404            for _ in 0..5 {
405                if std_dev > 0 {
406                    std_dev = (std_dev + variance_avg / std_dev) / 2;
407                }
408            }
409
410            // Filter outliers (beyond threshold standard deviations)
411            prices.iter()
412                .filter(|p| {
413                    let diff = if p.price > mean { p.price - mean } else { mean - p.price };
414                    diff <= std_dev * self.outlier_threshold as u128
415                })
416                .cloned()
417                .collect()
418        }

426        pub fn calculate_confidence_score(&self, prices: &[PriceData]) -> Result<u32, OracleError> {
427            if prices.is_empty() {
428                return Ok(0);
429            }
430
431            // Simple confidence based on number of sources and price variance
432            let source_confidence = (prices.len() as u32 * 25).min(75); // Max 75 from sources
433
434            // Calculate coefficient of variation
435            let sum: u128 = prices.iter().map(|p| p.price).sum();
436            let mean = sum / prices.len() as u128;
437
438            let variance: u128 = prices.iter()
439                .map(|p| {
440                    let diff = if p.price > mean { p.price - mean } else { mean - p.price };
441                    diff * diff
442                })
443                .sum();
444
445            // Calculate coefficient of variation using fixed point arithmetic
446            let std_dev = if prices.len() > 0 {
447                let variance_avg = variance / prices.len() as u128;
448                // Simple approximation of square root (for fixed point)
449                let mut approx = variance_avg;
450                for _ in 0..5 { // Newton-Raphson approximation
451                    if approx > 0 {
452                        approx = (approx + variance_avg / approx) / 2;
453                    }
454                }
455                approx
456            } else {
457                0
458            };
459
460            let cv = if mean > 0 {
461                (std_dev * 10000) / mean // Multiply by 10000 for precision
462            } else {
463                10000
464            };
465
466            // Lower CV = higher confidence (CV is in basis points)
467            let variance_confidence = if cv <= 10000 {
468                ((10000 - cv) / 400) as u32 // Scale to 0-25 range
469            } else {
470                0
471            };
472
473            Ok(source_confidence + variance_confidence)
474        }

476        fn calculate_volatility(&self, property_id: u64) -> Result<u32, OracleError> {
477            let historical = self.get_historical_valuations(property_id, 30); // Last 30 valuations
478
479            if historical.len() < 2 {
480                return Ok(0);
481            }
482
483            // Calculate price changes
484            let mut changes = Vec::new();
485            for i in 1..historical.len() {
486                let prev = historical[i-1].valuation;
487                let curr = historical[i].valuation;
488
489                if prev > 0 {
490                    let change = if curr > prev {
491                        ((curr - prev) * 10000) / prev // Change in basis points
492                    } else {
493                        ((prev - curr) * 10000) / prev
494                    };
495                    changes.push(change);
496                }
497            }
498
499            // Average absolute change as volatility index (in basis points)
500            let total_change: u128 = changes.iter().sum();
501            let avg_change_bp = total_change / changes.len() as u128;
502            Ok((avg_change_bp / 100).min(100) as u32) // Convert to percentage
503        }

505        fn calculate_confidence_interval(&self, valuation: &PropertyValuation) -> Result<(u128, u128), OracleError> {
506            // Simple confidence interval based on confidence score
507            let margin = valuation.valuation * (100 - valuation.confidence_score) as u128 / 10000; // 1% per confidence point
508
509            Ok((
510                valuation.valuation.saturating_sub(margin),
511                valuation.valuation + margin
512            ))
513        }

534        fn check_price_alerts(&mut self, property_id: u64, new_valuation: u128) -> Result<(), OracleError> {
535            if let Some(last_valuation) = self.property_valuations.get(&property_id) {
536                let change_percentage = self.calculate_percentage_change(last_valuation.valuation, new_valuation);
537
538                if let Some(alerts) = self.price_alerts.get(&property_id) {
539                    for alert in alerts {
540                        if alert.is_active && change_percentage >= alert.threshold_percentage as u128 {
541                            self.env().emit_event(PriceAlertTriggered {
542                                property_id,
543                                old_valuation: last_valuation.valuation,
544                                new_valuation,
545                                change_percentage: change_percentage as u32,
546                                alert_address: alert.alert_address,
547                            });
548                        }
549                    }
550                }
551            }
552            Ok(())
553        }

555        pub fn calculate_percentage_change(&self, old_value: u128, new_value: u128) -> u128 {
556            if old_value == 0 {
557                return 0;
558            }
559
560            let diff = if new_value > old_value {
561                new_value - old_value
562            } else {
563                old_value - new_value
564            };
565
566            (diff * 100) / old_value
567        }
